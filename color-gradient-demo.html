Implementation Functions

// Option 1: Saturation Gradient
function getColorBySaturation(percentChange) {
  const absChange = Math.abs(percentChange);
  const intensity = Math.min(absChange * 10, 80); // Cap at 80% opacity
  
  if (percentChange > 0) {
    return `rgba(34, 197, 94, ${intensity/100})`; // Green with variable opacity
  } else if (percentChange < 0) {
    return `rgba(220, 38, 38, ${intensity/100})`; // Red with variable opacity  
  } else {
    return `rgba(107, 114, 128, 0.3)`; // Neutral gray
  }
}

// Option 2: Temperature Gradient  
function getTemperatureColor(percentChange) {
  if (Math.abs(percentChange) < 0.1) return 'rgba(107, 114, 128, 0.3)';
  
  const intensity = Math.min(Math.abs(percentChange) * 8, 80);
  
  if (percentChange > 0) {
    return `rgba(251, 146, 60, ${intensity/100})`; // Orange/warm
  } else {
    return `rgba(59, 130, 246, ${intensity/100})`; // Blue/cool
  }
}

// Option 3: Terminal Style
function getTerminalColor(percentChange) {
  const absChange = Math.abs(percentChange);
  if (absChange < 0.1) return { bg: 'rgba(107, 114, 128, 0.1)', text: '#9ca3af' };
  
  const intensity = Math.min(absChange * 10, 100);
  
  if (percentChange > 0) {
    return { 
      bg: `rgba(0, 255, 65, ${intensity/1000})`, 
      text: `rgba(0, 255, 65, ${intensity/100})` 
    };
  } else {
    return { 
      bg: `rgba(255, 0, 128, ${intensity/1000})`, 
      text: `rgba(255, 0, 128, ${intensity/100})` 
    };
  }
}
